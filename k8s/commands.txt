# Commandes pour créer les resources Kubernetes 

# === NAMESPACE ===
kubectl create namespace potioncraft --dry-run=client -o yaml > namespace.yaml

# Commandes pour créer les resources Kubernetes

# === WORDS (API) ===
kubectl create deployment api --image=docker.io/rekimete/potioncraft-api -n potioncraft --dry-run=client -o yaml > api-deployment.yaml
kubectl create service clusterip api --tcp=80:80 -n potioncraft --dry-run=client -o yaml > api-service.yaml

# === WEB (Nginx) ===
kubectl create deployment frontend --image=docker.io/rekimete/potioncraft-frontend --port=80 -n potioncraft --dry-run=client -o yaml > frontend-deployment.yaml
kubectl create service clusterip frontend --tcp=80:80 -n potioncraft --dry-run=client -o yaml > frontend-service.yaml

# === APPLY ===
kubectl apply -f namespace.yaml
kubectl apply -f .

# === PORT FORWARD (pour accès local) ===
kubectl port-forward -n potioncraft svc/frontend 8080:80
kubectl port-forward -n potioncraft svc/api 8081:80

# === DELETE ===
kubectl delete -f .
